\def\tm@littlehalf{
  \pic[scale=0.5,color/.expanded=\tm@color] {tm-half-note-head-center};
  \draw[scale=0.5,color/.expanded=\tm@color] (\tm@notewidth,0) -- ++ (0,.6);
}
\newcommand\tmlittlehalf[1][]{\begingroup\tmset{#1}\tikz{\tm@littlehalf}\endgroup}
\newcommand\tmlittlehalfdot[1][]{%
  \begingroup\tmset{#1}\tikz{
    \tm@littlehalf\fill[color/.expanded=\tm@color] (.15,0) circle (.8pt);
  }\endgroup%
}
