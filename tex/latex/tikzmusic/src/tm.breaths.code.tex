\newcommand\tmbreath[2][]{
  \begingroup
    \tmset{#1}
    \pic[color/.expanded=\tm@color,shift={(0,.4)}] at 
      ([shift={(\tm@shift@start@x,\tm@shift@start@y)}]#2,\tm@lineshift*.1) {tm-breath-mark};
  \endgroup
}
\newcommand\tmcaesura[2][]{
  \begingroup
    \tmset{#1}
    \fill[color/.expanded=\tm@color,shift={($(#2,\tm@lineshift*.1)+(\tm@shift@start@x,\tm@shift@start@y)$)}] 
      (-.3,.2) -- (-.2,.2) -- (.1,.6) -- (0,.6) -- cycle;
    \fill[color/.expanded=\tm@color,shift={($(#2,\tm@lineshift*.1)+(\tm@shift@start@x,\tm@shift@start@y)$)}] 
      (-.1,.2) -- (0,.2) -- (.3,.6) -- (.2,.6) -- cycle;
  \endgroup
}
